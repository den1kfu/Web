Програмування (C++)

Приклад коду, який знаходить суму елементів масиву:


#include <iostream>
using namespace std;

int main() {
    int numbers[] = {10, 20, 30, 40, 50};
    int sum = 0;

    
    for (int i : numbers) {
        sum += i;
    }

    cout << "Сума елементів: " << sum << endl;
    return 0;
}
